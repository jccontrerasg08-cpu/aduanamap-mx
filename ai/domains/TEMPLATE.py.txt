from __future__ import annotations

import re

from .. import knowledge

NAME = "mi_tema"                 # unique id, snake_case
SEED = "mi_tema.json"            # file in data/seed/
PRIORITY = 50                    # optional: lower matches first

# Keep the matcher cheap and specific — it must not swallow unrelated questions.
_PAT = re.compile(r"\b(palabra_clave|otra_clave)\b", re.IGNORECASE)


def match(q: str) -> bool:
    return bool(_PAT.search(q or ""))


def answer(q: str, lang: str = "es") -> dict:
    doc = knowledge.load_seed(SEED)          # {} if the seed is missing
    entries = doc.get("entries", [])
    if not entries:
        return {
            "answer": ("Aún no tengo cargada esa información."
                       if lang != "en" else "I don't have that information loaded yet."),
            "warnings": ["no confirmable: seed no disponible"],
        }

    # …find the relevant entry/entries and build prose from them only…
    hit = entries[0]
    text = f"{hit['term']}: {hit['definition']}"

    return {
        "answer": text,
        "data": {"entry": hit},
        # Cite the real-world source, taken from the seed entry.
        "sources": [{"source": hit.get("source_name", "SNICE"), "label": NAME}],
        "warnings": [],
    }
